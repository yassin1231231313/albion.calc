    <script>
        // Custom Local Database of materials with default base prices
        const USER_PRICES = {
            wood: 350,
            metal: 420,
            leather: 390,
            cloth: 310,
            crest_lymhurst: 25000,
            crest_thetford: 22000,
            crest_bridgewatch: 24000
        };

        const CATEGORY_MAP = {
            "Refining": [
                { name: "Plank Refining", id: "PLANKS", recipe: { wood: 4 } },
                { name: "Steel Refining", id: "METALBAR", recipe: { metal: 4 } },
                { name: "Leather Refining", id: "LEATHER", recipe: { leather: 4 } },
                { name: "Cloth Refining", id: "CLOTH", recipe: { cloth: 4 } }
            ],
            "Weapons · Warrior": [
                { name: "Swords", id: "MAIN_SWORD", recipe: { metal: 16, leather: 8 } },
                { name: "Axes", id: "MAIN_AXE", recipe: { metal: 12, wood: 12 } }
            ],
            "Accessories": [
                { name: "Bags, Satchels & Tracking", id: "BAG", recipe: { leather: 8, cloth: 8 } },
                { name: "Capes", id: "CAPE", recipe: { cloth: 4, leather: 4, crest_lymhurst: 1 } }
            ],
            "Consumables": [
                { name: "Soups", id: "MEAL_SOUP", recipe: { cloth: 0 }, staticCost: 4500 },
                { name: "Salads", id: "MEAL_SALAD", recipe: { cloth: 0 }, staticCost: 6500 }
            ]
        };

        let currentActiveSection = "home";
        let currentTargetCalculator = "";

        function navigateTo(viewId, payload = "") {
            // Hide all panels
            document.querySelectorAll('.view-panel').forEach(panel => panel.classList.add('hidden'));
            
            // Show target panel
            document.getElementById(`view-${viewId}`).classList.remove('hidden');

            // Manage sidebar selection states
            document.querySelectorAll('.nav-btn').forEach(btn => {
                btn.classList.remove('text-white', 'bg-gradient-to-r', 'from-[#201d18]', 'to-transparent', 'border-l-2', 'border-amber-500');
                btn.classList.add('text-gray-400');
            });

            // Set up route states
            if (viewId === 'home') {
                currentActiveSection = "home";
            } else if (viewId === 'material-prices') {
                currentActiveSection = "prices";
                renderMaterialPrices();
            } else if (viewId === 'category-menu') {
                document.getElementById('category-menu-title').innerText = payload;
                const menuList = document.getElementById('category-menu-list');
                menuList.innerHTML = "";
                
                const subs = CATEGORY_MAP[payload] || [];
                subs.forEach(sub => {
                    const card = document.createElement('div');
                    card.className = "bg-[#0f1116] border border-[#1b1e26] hover:border-amber-500/50 rounded-lg p-6 cursor-pointer transition text-center";
                    card.onclick = () => navigateTo('calculator', sub.name);
                    card.innerHTML = `
                        <div class="text-white font-medium text-sm mb-1">${sub.name}</div>
                        <div class="text-[10px] text-gray-500">View configuration and recipes</div>
                    `;
                    menuList.appendChild(card);
                });
            } else if (viewId === 'calculator') {
                currentTargetCalculator = payload;
                document.getElementById('calculator-title').innerText = payload;
                recalculateCurrent();
            }
        }

        function renderMaterialPrices() {
            const grid = document.getElementById('custom-prices-grid');
            grid.innerHTML = "";
            Object.keys(USER_PRICES).forEach(key => {
                const itemDiv = document.createElement('div');
                itemDiv.className = "bg-[#0f1116] border border-[#1b1e26] rounded p-4";
                itemDiv.innerHTML = `
                    <label class="block text-[10px] font-bold text-gray-500 uppercase mb-2">${key.replace('_', ' ')}</label>
                    <input type="number" oninput="USER_PRICES['${key}'] = parseFloat(this.value) || 0" value="${USER_PRICES[key]}" class="w-full bg-[#151821] border border-[#242936] text-white rounded p-2 text-xs focus:outline-none">
                `;
                grid.appendChild(itemDiv);
            });
        }

        function recalculateCurrent() {
            const rrrSelect = parseFloat(document.getElementById('calc-location').value);
            const rrrBonus = parseFloat(document.getElementById('calc-bonus-day').value);
            const focusChecked = document.getElementById('calc-focus').checked;
            const fee = parseFloat(document.getElementById('calc-fee').value) || 0;

            // Final combined RRR math
            let finalRrr = rrrSelect + rrrBonus;
            if (focusChecked) finalRrr = 0.59; // Focus overrides standard city metrics

            document.getElementById('calculated-rrr').innerText = `${Math.round(finalRrr * 1000) / 10}%`;

            let currentItemData = null;
            Object.keys(CATEGORY_MAP).forEach(cat => {
                const match = CATEGORY_MAP[cat].find(item => item.name === currentTargetCalculator);
                if (match) currentItemData = match;
            });

            if (!currentItemData) return;

            const tableBody = document.getElementById('calc-table-rows');
            tableBody.innerHTML = "";

            // Render Tiers 4 to 8
            for (let tier = 4; tier <= 8; tier++) {
                const tr = document.createElement('tr');
                tr.className = "hover:bg-[#15171e]/50 transition border-b border-[#1b1e26]";

                let cellsHtml = `<td class="p-4 font-bold text-white text-xs bg-[#15171e]/20 w-1/4">Tier ${tier}</td>`;

                // Render Enchantments .0 to .3
                for (let ench = 0; ench <= 3; ench++) {
                    let totalProductionCost = 0;

                    if (currentItemData.staticCost) {
                        totalProductionCost = currentItemData.staticCost * (tier - 2) * (1 + (ench * 0.2));
                    } else {
                        // Math recipe loop with RRR applied
                        Object.keys(currentItemData.recipe).forEach(matKey => {
                            const quantity = currentItemData.recipe[matKey];
                            const basePrice = USER_PRICES[matKey] || 300;
                            const levelMultiplier = 1 + (ench * 0.25) + ((tier - 4) * 0.15);
                            totalProductionCost += (quantity * (1 - finalRrr)) * (basePrice * levelMultiplier);
                        });
                        totalProductionCost += fee * (tier / 2);
                    }

                    cellsHtml += `
                        <td class="p-4 text-center font-mono text-gray-300">
                            ${Math.round(totalProductionCost).toLocaleString()} <span class="text-[10px] text-gray-500">Silver</span>
                        </td>
                    `;
                }

                tr.innerHTML = cellsHtml;
                tableBody.appendChild(tr);
            }
        }

        function resetApp() {
            document.getElementById('calc-location').value = "0.152";
            document.getElementById('calc-bonus-day').value = "0.0";
            document.getElementById('calc-focus').checked = false;
            document.getElementById('calc-fee').value = "1000";
            recalculateCurrent();
        }
    </script>
