    <script>
        // CORE DB TEMPLATES - Used to generate dozens of physical items across different Tiers and Enchantments dynamically
        const ITEM_TEMPLATES = [
            // WEAPONS
            { baseId: "MAIN_SPEAR", name: "Spear", type: "weapons", nutrition: 180, recipe: { wood: 8, metal: 16 } },
            { baseId: "MAIN_BOW", name: "Bow", type: "weapons", nutrition: 180, recipe: { wood: 32 } },
            { baseId: "MAIN_SWORD", name: "Broadsword", type: "weapons", nutrition: 180, recipe: { metal: 16, leather: 8 } },
            { baseId: "MAIN_CURSESTAFF", name: "Cursed Staff", type: "weapons", nutrition: 180, recipe: { wood: 16, cloth: 8 } },
            
            // ARMOR
            { baseId: "HEAD_PLATE_SET1", name: "Soldier Helmet", type: "armor", nutrition: 90, recipe: { metal: 4 } },
            { baseId: "ARMOR_PLATE_SET1", name: "Soldier Armor", type: "armor", nutrition: 180, recipe: { metal: 16 } },
            { baseId: "SHOES_PLATE_SET1", name: "Soldier Boots", type: "armor", nutrition: 90, recipe: { metal: 4 } },
            { baseId: "ARMOR_LEATHER_SET1", name: "Mercenary Jacket", type: "armor", nutrition: 180, recipe: { leather: 16 } },
            
            // CONSUMABLES
            { baseId: "MEAL_SOUP", name: "Soup (Cabbage T5)", type: "consumables", customTier: 5, nutrition: 284, recipe: { T5_CABBAGE: 48, T4_BUTTER: 8 } },
            { baseId: "MEAL_STEW", name: "Stew (Beef T8)", type: "consumables", customTier: 8, nutrition: 320, recipe: { T8_BEEF: 48, T6_BUTTER: 8 } },
            { baseId: "POTION_HEAL", name: "Healing Potion", type: "consumables", customTier: 4, nutrition: 45, recipe: { T4_HERB_SHIELD_LICHEN: 24, T3_HERB_VALERIAN: 12 } },
            
            // CAPES
            { baseId: "CAPE", name: "Regular Cape", type: "capes", nutrition: 45, recipe: { cloth: 4, leather: 4 } }
        ];

        let currentFilter = "all";

        function filterCategory(cat) {
            currentFilter = cat;
            document.querySelectorAll('.category-tab').forEach(btn => {
                btn.classList.remove('bg-amber-600', 'text-white');
                btn.classList.add('bg-[#252836]', 'text-gray-300');
            });
            event.target.classList.add('bg-amber-600', 'text-white');
            runCalculationEngine();
        }

        // Helper to dynamically build material IDs based on Tier & Enchantment
        function getMaterialId(matType, tier, ench) {
            const prefix = `T${tier}_`;
            let suffix = "";
            if (matType === "wood") suffix = "PLANKS";
            if (matType === "metal") suffix = "METALBAR";
            if (matType === "leather") suffix = "LEATHER";
            if (matType === "cloth") suffix = "CLOTH";
            
            const baseMat = prefix + suffix;
            return ench > 0 ? `${baseMat}_LEVEL${ench}` : baseMat;
        }

        async function runCalculationEngine() {
            const server = document.getElementById("serverSelect").value;
            const targetCity = document.getElementById("marketCity").value;
            const rrr = parseFloat(document.getElementById("useFocus").checked ? 0.479 : document.getElementById("rrrSelect").value);
            const fee = parseFloat(document.getElementById("stationFee").value) || 0;

            const finalCalculatedList = [];
            const idsToFetch = new Set();

            // Generate full inventory lists (Across Tiers 4, 5, 6, 7 & 8 and Enchantments 0, 1, 2)
            ITEM_TEMPLATES.forEach(item => {
                if (currentFilter !== "all" && item.type !== currentFilter) return;

                // Handle items with strict static recipes (Foods & Potions)
                if (item.customTier) {
                    const itemId = `T${item.customTier}_${item.baseId}`;
                    idsToFetch.add(itemId);
                    Object.keys(item.recipe).forEach(ingId => idsToFetch.add(ingId));
                    finalCalculatedList.push({
                        name: item.name,
                        id: itemId,
                        tier: `T${item.customTier}.0`,
                        recipe: item.recipe,
                        nutrition: item.nutrition,
                        isStatic: true
                    });
                    return;
                }

                // Handle scaleable gear (Weapons, Armor, Capes)
                for (let tier = 4; tier <= 7; tier++) {
                    for (let ench = 0; ench <= 2; ench++) {
                        const baseItemId = `T${tier}_${item.baseId}`;
                        const finishedProductId = ench > 0 ? `${baseItemId}@${ench}` : baseItemId;
                        idsToFetch.add(finishedProductId);

                        // Resolve material ingredient lists
                        const ingredientsList = {};
                        Object.keys(item.recipe).forEach(matType => {
                            const matId = getMaterialId(matType, tier, ench);
                            ingredientsList[matId] = item.recipe[matType];
                            idsToFetch.add(matId);
                        });

                        finalCalculatedList.push({
                            name: item.name,
                            id: finishedProductId,
                            tier: `T${tier}.${ench}`,
                            recipe: ingredientsList,
                            nutrition: item.nutrition,
                            isStatic: false
                        });
                    }
                }
            });

            // Call API
            const idsParam = Array.from(idsToFetch).join(",");
            const apiUrl = `https://${server}.albion-online-data.com/api/v2/stats/prices/${idsParam}.json?locations=${targetCity}`;

            const tbody = document.getElementById("masterTableBody");
            tbody.innerHTML = `<tr><td colspan="6" class="px-6 py-12 text-center text-amber-500 animate-pulse font-medium">Downloading market data logs... Please wait...</td></tr>`;

            try {
                const response = await fetch(apiUrl);
                const priceData = await response.json();

                // Map results for lightning-fast lookups
                const priceMap = {};
                priceData.forEach(p => {
                    priceMap[p.item_id] = p.sell_price_min || p.buy_price_max || 0;
                });

                renderMasterTable(finalCalculatedList, priceMap, rrr, fee);
            } catch (err) {
                tbody.innerHTML = `<tr><td colspan="6" class="px-6 py-12 text-center text-red-500">Failed to pull real-time API logs. Check if the selected server is responsive.</td></tr>`;
                console.error(err);
            }
        }

        function renderMasterTable(calculatedItems, priceMap, rrr, fee) {
            const tbody = document.getElementById("masterTableBody");
            tbody.innerHTML = "";

            if (calculatedItems.length === 0) {
                tbody.innerHTML = `<tr><td colspan="6" class="px-6 py-12 text-center text-gray-500">No items match this filter category.</td></tr>`;
                return;
            }

            calculatedItems.forEach(item => {
                let ingredientCost = 0;
                let missingPrice = false;

                Object.keys(item.recipe).forEach(ingId => {
                    const price = priceMap[ingId] || 0;
                    if (price === 0) missingPrice = true;
                    const amount = item.recipe[ingId];
                    ingredientCost += (amount * (1 - rrr)) * price;
                });

                const stationFee = (item.nutrition / 100) * fee;
                const totalCost = Math.round(ingredientCost + stationFee);
                const marketValue = priceMap[item.id] || 0;
                
                // 4% setup fee + tax deductions
                const marketTaxes = marketValue * 0.04; 
                const netProfit = marketValue > 0 ? Math.round(marketValue - totalCost - marketTaxes) : 0;
                const profitClass = netProfit > 0 ? "text-green-400 font-bold" : (netProfit < 0 ? "text-red-400" : "text-gray-500");

                const row = document.createElement("tr");
                row.className = "hover:bg-[#1f212a] border-b border-gray-800 transition-colors";
                row.innerHTML = `
                    <td class="px-6 py-4 flex items-center gap-3">
                        <img class="w-10 h-10 rounded border border-gray-700 bg-[#252836]" src="https://render.albiononline.com/v1/item/${item.id}.png" alt="" onerror="this.src='https://placehold.co/40x40/252836/ffffff?text=Item'">
                        <div>
                            <div class="text-white font-medium">${item.name}</div>
                            <div class="text-xs text-gray-500 font-mono select-all">${item.id}</div>
                        </div>
                    </td>
                    <td class="px-6 py-4">
                        <span class="bg-[#2a2d3d] text-amber-500 px-2 py-1 rounded text-xs font-mono font-bold">${item.tier}</span>
                    </td>
                    <td class="px-6 py-4 font-mono text-sm text-gray-300">
                        ${missingPrice ? '<span class="text-xs text-amber-500/70">⚠️ Low Market Logs</span>' : `${Math.round(ingredientCost).toLocaleString()} Silver`}
                    </td>
                    <td class="px-6 py-4 font-mono text-xs text-gray-400">
                        ${Math.round(stationFee).toLocaleString()} Silver
                    </td>
                    <td class="px-6 py-4 font-mono text-sm text-gray-300">
                        ${marketValue > 0 ? `${marketValue.toLocaleString()} Silver` : '<span class="text-xs text-gray-500">No Sale Listings</span>'}
                    </td>
                    <td class="px-6 py-4 font-mono text-sm ${profitClass}">
                        ${marketValue === 0 ? '-' : (netProfit > 0 ? `+${netProfit.toLocaleString()}` : netProfit.toLocaleString())} Silver
                    </td>
                `;
                tbody.appendChild(row);
            });
        }
    </script>
